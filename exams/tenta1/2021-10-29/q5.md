int main() {

    unsigned int k {}; // Tiden kommer alltid vara postitiv
    std::array<k, int> wats {}; // Vi vet hur många element vi behöver spara, därför kan vi använda en array. Det kommer då alltid vara O(1) insättning.
    int sum _of_wats = 0;
    double average {};

    void get_wats(int x) { // =O(k) i tidskomplex, men även O(k) i minneshantering

        wats.push_back(x);

        sum_of_wats += x
        average = sum_of_wats / (int)wats.size();

        time.sleep(1); // Inte rätt syntax, men programmet pausar en gång i sekunden. - Vi vill köra denna loop 1 sek i sekunden
    }

    return 0;
}
